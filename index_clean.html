        let totalTons = 0;

        let totalCredits = 0;

        let recyclableItems = 0;

        let totalScans = 0;

       

        const updateImpactMetrics = (isRecyclable) => {

            totalTons += 0.01; // Simulate 10 kg per scan

            totalScans += 1;

            if (isRecyclable) {

                recyclableItems += 1;

                totalCredits += 0.005; // Simulate 5 kg per recyclable scan

        const loadImageToCanvas = (src) => {

            return new Promise((resolve, reject) => {

                const img = new Image();

                img.crossOrigin = 'anonymous'; // Important for fetching from other domains

                img.onload = () => {

                    const ctx = imageCanvas.getContext('2d');

                    const MAX_WIDTH = 500;

                    const scale = MAX_WIDTH / img.width;

                    imageCanvas.width = MAX_WIDTH;

                    imageCanvas.height = img.height * scale;

                    ctx.drawImage(img, 0, 0, imageCanvas.width, imageCanvas.height);

                    placeholderText.classList.add('hidden');

                    predictBtn.classList.remove('hidden');

                    resultsContainer.classList.add('hidden');

                    resultsList.innerHTML = '';

                    resolve();

                };

                img.onerror = () => {

                    reject(new Error('Failed to load image from URL.'));

                };

                img.src = src;

            });

        };

 

        imageUpload.addEventListener('change', event => {

            const file = event.target.files[0];

            if (!file) return;

 

            const reader = new FileReader();

            reader.onload = e => {

                    // Using an alert for simplicity in this single-file context

                    alert("Error loading file: " + error.message);

                    placeholderText.classList.remove('hidden');

                });

            };

            reader.readAsDataURL(file);

        });

                placeholderText.classList.remove('hidden');
